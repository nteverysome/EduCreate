addWord() {
  const word = {
    id: Date.now() + Math.random(),
    english: this.newWord.english.trim(),
    chinese: this.newWord.chinese.trim(),
    createdAt: new Date()
  };
  this.vocabulary.push(word);
  this.autoSave();

// 游戏选择功能
selectGame(game) {
  if (this.vocabulary.length < game.minWords) {
    this.showSuccess(`${game.name} 至少需要 ${game.minWords} 个词汇！`);
    return;
  }
  this.selectedGame = game;
}
